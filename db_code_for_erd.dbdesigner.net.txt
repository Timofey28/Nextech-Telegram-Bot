
clients {
	id serial
	surname varchar(30)
	name varchar(30)
	patronymic varchar(30)
	sex varchar(6)
	tg_id bigint
	inn varchar(15)
	business_direction_id int *> business_directions.id
	phone_number varchar(11)
	email varchar(50)
}

sent_messages {
	id serial
	tg_id bigint *> clients.tg_id
	message_id int
	sending_datetime timestamp
}

business_directions {
	id serial
	name varchar(50)
}
