
clients {
	id serial
	surname varchar(30)
	name varchar(30)
	patronymic varchar(30)
	sex varchar(6)
	tg_id bigint
	inn varchar(15)
	business_direction_id int *> business_directions.id
	phone_number varchar(11)
	email varchar(50)
}

sent_messages {
	id serial
	tg_id bigint *> clients.tg_id
	message_id int
	sending_datetime timestamp
}

business_directions {
	id serial
	name varchar(50)
}

employees {
	id serial
	first_last_name varchar(60)
	phone_number varchar(11)
}

shifts {
	empleyee_id int *> employees.id
	product_barcode varchar(50)
	product_name varchar(70)
	tariff_price int
	amount int
	date_ date
	act_number int
	was_paid boolean
}
